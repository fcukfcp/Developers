//iterative
#include <stdio.h>
#include <conio.h>

//prototype declaration
int factorial(int );

int factorial(int n){
int product=1;
while(n>1){
  product*=n;
   n-- ;
 }
return product;
}

int main(){
int number ,result;
clrscr();
printf("Enter a number :\n");
scanf(" %d",&number);
result=factorial(number);
printf("Factorial %d=%d",number,result);
getch();
return 0;
}

#include<stdio.h>
#include<conio.h>
//prototype declaration
int swap(int,int);
int swap(int *a , int *b){
int temp;
temp=*a;
*a=*b;
*b=temp;
}
int main(){
int num1,num2 ;
clrscr();
printf("Enter two number");
scanf(" %d %d",&num1,&num2);
printf("After swapping ");
swap(&num1,&num2);
printf("Num1=%d Num2=%d",num1,num2);
return 0;
getch();
}
